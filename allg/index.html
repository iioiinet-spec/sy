Chapter 01. 최댓값 찾기

배열 내부의 원소를 순차 탐색하며 최대 포인터를 갱신하는 기본 구조입니다. C 언어에서는 원소의 수(n)를 매개변수로 명시해 주어야 합니다.

max_finder.cC99
#include <stdio.h>

int find_max(int arr[], int n) {
    int max_val = arr[0]; // 초기 첫 번째 원소를 최댓값으로 설정
    for (int i = 1; i < n; i++) {
        if (arr[i] > max_val) {
            max_val = arr[i]; // 더 큰 값 발견 시 갱신
        }
    }
    return max_val;
}

int main() {
    int scores[] = {34, 78, 12, 99, 56};
    int n = sizeof(scores) / sizeof(scores[0]);
    printf("최댓값: %d\n", find_max(scores, n)); // 출력: 99
    return 0;
}

Chapter 03. 정렬 알고리즘

다양한 기본/고급 정렬 알고리즘의 C 언어 구현입니다. 고급 정렬인 합병, 퀵 정렬은 동적 할당 오버헤드를 막기 위해 인덱스 기반의 포인터 방식을 취합니다.

1) 버블 정렬 (Bubble Sort)

bubble_sort.c
void bubble_sort(int arr[], int n) {
    for (int i = 0; i < n - 1; i++) {
        int swapped = 0;
        for (int j = 0; j < n - i - 1; j++) {
            if (arr[j] > arr[j + 1]) {
                int temp = arr[j];
                arr[j] = arr[j + 1];
                arr[j + 1] = temp;
                swapped = 1;
            }
        }
        if (!swapped) break; // 최적화: 정렬이 이미 끝났다면 종료
    }
}

2) 선택 정렬 (Selection Sort)

selection_sort.c
void selection_sort(int arr[], int n) {
    for (int i = 0; i < n - 1; i++) {
        int min_idx = i;
        for (int j = i + 1; j < n; j++) {
            if (arr[j] < arr[min_idx]) min_idx = j;
        }
        // 교환(Swap)
        int temp = arr[i];
        arr[i] = arr[min_idx];
        arr[min_idx] = temp;
    }
}

3) 합병 정렬 (Merge Sort)

merge_sort.c
int sorted[100005]; // 병합을 위한 전역 임시 배열

void merge(int arr[], int left, int mid, int right) {
    int i = left, j = mid + 1, k = left;
    while (i <= mid && j <= right) {
        if (arr[i] <= arr[j]) sorted[k++] = arr[i++];
        else sorted[k++] = arr[j++];
    }
    while (i <= mid) sorted[k++] = arr[i++];
    while (j <= right) sorted[k++] = arr[j++];
    for (int l = left; l <= right; l++) arr[l] = sorted[l];
}

void merge_sort(int arr[], int left, int right) {
    if (left < right) {
        int mid = (left + right) / 2;
        merge_sort(arr, left, mid);
        merge_sort(arr, mid + 1, right);
        merge(arr, left, mid, right);
    }
}

4) 퀵 정렬 (Quick Sort)

quick_sort.c
void quick_sort(int arr[], int left, int right) {
    if (left >= right) return;
    int pivot = arr[(left + right) / 2];
    int i = left, j = right;
    
    while (i <= j) {
        while (arr[i] < pivot) i++;
        while (arr[j] > pivot) j--;
        if (i <= j) {
            int temp = arr[i]; arr[i] = arr[j]; arr[j] = temp;
            i++; j--;
        }
    }
    quick_sort(arr, left, j);
    quick_sort(arr, i, right);
}

Chapter 04. 탐색 알고리즘

순차 탐색, 이진 탐색 및 그래프 전수 조사를 위한 DFS와 BFS의 전형적인 C 기법군입니다. 가변 큐/리스트 라이브러리가 없으므로 정적 배열로 연결리스트와 전역 큐를 구성합니다.

1) 이진 탐색 (Binary Search)

binary_search.c
int binary_search(int arr[], int n, int target) {
    int left = 0, right = n - 1;
    while (left <= right) {
        int mid = (left + right) / 2;
        if (arr[mid] == target) return mid;
        else if (arr[mid] < target) left = mid + 1;
        else right = mid - 1;
    }
    return -1;
}

2) DFS & BFS (인접 리스트 구조 표현)

graph_search.c
#include <stdio.h>

int graph[101][101]; // graph[i][j]: 노드 i의 j번째 이웃
int degree[101];     // degree[i]: 노드 i에 연결된 간선 수
int visited[101];

void dfs(int node) {
    visited[node] = 1;
    printf("%d ", node);
    for (int i = 0; i < degree[node]; i++) {
        int next = graph[node][i];
        if (!visited[next]) dfs(next);
    }
}

int queue[10005];
void bfs(int start) {
    int front = 0, rear = 0;
    queue[rear++] = start;
    visited[start] = 1;
    
    while (front < rear) {
        int curr = queue[front++];
        printf("%d ", curr);
        for (int i = 0; i < degree[curr]; i++) {
            int next = graph[curr][i];
            if (!visited[next]) {
                visited[next] = 1;
                queue[rear++] = next;
            }
        }
    }
}

3) 유니온-파인드 (Union-Find)

union_find.c
int parent[1005];

void init_set(int n) {
    for (int i = 1; i <= n; i++) parent[i] = i;
}

int find_root(int i) {
    if (parent[i] == i) return i;
    return parent[i] = find_root(parent[i]); // 경로 압축 (Path Compression)
}

void union_elements(int i, int j) {
    int root_i = find_root(i);
    int root_j = find_root(j);
    if (root_i != root_j) parent[root_i] = root_j;
}

Chapter 06. 탐욕 알고리즘 (Greedy)

매 순간 가장 이익이 큰 선택을 영리하게 취하는 그리디 기법의 전형인 거스름돈 반환 구조식입니다.

coin_change.c
#include <stdio.h>

int main() {
    int coins[] = {500, 100, 50, 10};
    int change = 1260;
    int total_count = 0;
    int num_types = sizeof(coins) / sizeof(coins[0]);

    for (int i = 0; i < num_types; i++) {
        total_count += (change / coins[i]);
        change %= coins[i];
    }
    printf("최소 동전 개수: %d\n", total_count); // 출력: 5 (500원 2개, 100원 2개, 50원 1개, 10원 1개)
    return 0;
}

Chapter 07. 동적 계획법 (Dynamic Programming)

한 번 연산된 중복 부분 문제를 메모이제이션 배열에 영구 저장해 놓고 연산 단계를 크게 줄이는 피보나치 상향식(Bottom-Up) 구조입니다.

fibonacci_dp.c
#include <stdio.h>

long long dp[91]; // 피보나치 수는 빠르게 증가하므로 long long 필수

long long get_fibonacci(int n) {
    dp[1] = 1;
    dp[2] = 1;
    for (int i = 3; i <= n; i++) {
        dp[i] = dp[i - 1] + dp[i - 2];
    }
    return dp[n];
}

int main() {
    int n = 50;
    printf("피보나치 %d번째 값: %lld\n", n, get_fibonacci(n));
    return 0;
}
