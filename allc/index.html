Chapter 01
알고리즘이란 무엇인가?
문제를 해결하는 단계적 절차 — 올림피아드의 모든 것은 여기서 시작됩니다
알고리즘의 정의

알고리즘(Algorithm)이란 어떤 문제를 해결하기 위한 명확하고 유한한 단계적 절차입니다. 알고리즘은 반드시 다음 5가지 조건을 갖춰야 합니다.

📥
입력 (Input)

0개 이상의 외부 입력이 존재해야 합니다

📤
출력 (Output)

1개 이상의 결과가 반드시 출력되어야 합니다

🔢
유한성 (Finiteness)

유한한 단계 후에 반드시 종료되어야 합니다

📋
명확성 (Definiteness)

각 단계가 명확하고 모호하지 않아야 합니다

⚡
효율성 (Effectiveness)

모든 연산은 실제로 수행 가능해야 합니다

좋은 알고리즘의 조건
⏱ 시간 효율성

같은 결과를 더 빠르게 계산할수록 좋습니다. 입력 크기 n이 커질 때 수행 시간이 얼마나 증가하는지가 핵심입니다.

💾 공간 효율성

더 적은 메모리를 사용할수록 좋습니다. 시간과 공간은 종종 트레이드오프(trade-off) 관계입니다.

✅ 정확성

모든 입력에 대해 올바른 결과를 내야 합니다. 빠르지만 틀린 알고리즘은 의미 없습니다.

📖 가독성

다른 사람이 이해하기 쉽게 작성되어야 합니다. 올림피아드에서도 코드의 명확성이 중요합니다.

첫 번째 예제 — 최댓값 찾기

배열에서 가장 큰 값을 찾는 가장 기본적인 알고리즘입니다.

알고리즘 최댓값 찾기(A, n) // A: 숫자 배열, n: 원소 수 max ← A[0] for i ← 1 to n-1 do if A[i] > max then max ← A[i] return max
최댓값 찾기 C 
#include <stdio.h>

int find_max(int arr[], int n) {
    int max_val = arr[0];              // 첫 원소를 임시 최댓값으로 저장

    for (int i = 1; i < n; i++) {
        if (arr[i] > max_val) {        // 더 큰 값을 발견하면 갱신
            max_val = arr[i];
        }
    }
    return max_val;
}

int main(void) {
    int arr[] = {3, 7, 1, 9, 4, 2};
    int n = sizeof(arr) / sizeof(arr[0]);

    printf("%d\n", find_max(arr, n));  // 출력: 9
    return 0;
}
🔍 분석

이 알고리즘은 배열의 모든 원소를 한 번씩 확인합니다. 원소 수가 n이면 최대 n번 비교 → 시간 복잡도 O(n)

⚠ 올림피아드 핵심 기준

1초에 약 1억(10⁸)번 연산 가능. n=10,000이면 O(n²)은 10억 번 → 시간 초과! O(n log n) 이하를 목표로 하세요.

Big-O 계산 방법
복잡도 계산 예시 C 
#include <stdio.h>

// O(1) — 입력 크기와 무관하게 항상 한 번에 접근
int get_first(int arr[]) {
    return arr[0];
}

// O(n) — 원소 n개를 한 번씩 확인
void print_all(int arr[], int n) {
    for (int i = 0; i < n; i++) {
        printf("%d ", arr[i]);
    }
    printf("\n");
}

// O(n²) — 이중 반복문
void print_pairs(int arr[], int n) {
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < n; j++) {
            printf("(%d, %d) ", arr[i], arr[j]);
        }
        printf("\n");
    }
}

// O(log n) — 매번 절반으로 줄임
int binary_search_count(int n) {
    int count = 0;
    while (n > 1) {
        n /= 2;
        count++;
    }
    return count;
}

① 버블 정렬 (Bubble Sort)

인접한 두 원소를 비교하여 큰 값을 오른쪽으로 밀어나가는 정렬입니다. O(n²)

🔍 동작 원리

한 번의 패스(pass)에서 가장 큰 원소가 맨 오른쪽으로 이동합니다. n-1번의 패스 후 정렬 완료.

예: [5, 3, 8, 1] → [3, 5, 1, 8] → [3, 1, 5, 8] → [1, 3, 5, 8]

버블 정렬C 
#include <stdio.h>

void bubble_sort(int arr[], int n) {
    for (int i = 0; i < n - 1; i++) {
        int swapped = 0;

        for (int j = 0; j < n - i - 1; j++) {
            if (arr[j] > arr[j + 1]) {
                int temp = arr[j];
                arr[j] = arr[j + 1];
                arr[j + 1] = temp;
                swapped = 1;
            }
        }

        if (!swapped) break;  // 교환이 없으면 이미 정렬 완료
    }
}
② 선택 정렬 (Selection Sort)

매 패스마다 최솟값을 찾아 맨 앞으로 이동시키는 정렬입니다. O(n²)

선택 정렬C 
#include <stdio.h>

void selection_sort(int arr[], int n) {
    for (int i = 0; i < n - 1; i++) {
        int min_idx = i;

        for (int j = i + 1; j < n; j++) {
            if (arr[j] < arr[min_idx]) {
                min_idx = j;
            }
        }

        int temp = arr[i];
        arr[i] = arr[min_idx];
        arr[min_idx] = temp;
    }
}
③ 합병 정렬 (Merge Sort)

분할정복을 사용하여 배열을 반으로 나누고 정렬 후 합병합니다. O(n log n) 항상 보장

합병 정렬C 
#include <stdio.h>

void merge(int arr[], int left, int mid, int right) {
    int temp[1000];
    int i = left;
    int j = mid + 1;
    int k = 0;

    while (i <= mid && j <= right) {
        if (arr[i] <= arr[j]) temp[k++] = arr[i++];
        else temp[k++] = arr[j++];
    }
    while (i <= mid) temp[k++] = arr[i++];
    while (j <= right) temp[k++] = arr[j++];

    for (int t = 0; t < k; t++) {
        arr[left + t] = temp[t];
    }
}

void merge_sort(int arr[], int left, int right) {
    if (left >= right) return;

    int mid = (left + right) / 2;
    merge_sort(arr, left, mid);
    merge_sort(arr, mid + 1, right);
    merge(arr, left, mid, right);
}
④ 퀵 정렬 (Quick Sort)

피벗(pivot)을 기준으로 작은 값은 왼쪽, 큰 값은 오른쪽으로 분할합니다. 평균 O(n log n), 최악 O(n²)

퀵 정렬C 
#include <stdio.h>

void quick_sort(int arr[], int left, int right) {
    if (left >= right) return;

    int pivot = arr[(left + right) / 2];
    int i = left;
    int j = right;

    while (i <= j) {
        while (arr[i] < pivot) i++;
        while (arr[j] > pivot) j--;

        if (i <= j) {
            int temp = arr[i];
            arr[i] = arr[j];
            arr[j] = temp;
            i++;
            j--;
        }
    }

    quick_sort(arr, left, j);
    quick_sort(arr, i, right);
}

① 선형 탐색 (Linear Search)

처음부터 끝까지 하나씩 확인합니다. O(n) — 정렬 불필요

선형 탐색C
#include <stdio.h>

int linear_search(int arr[], int n, int target) {
    for (int i = 0; i < n; i++) {
        if (arr[i] == target) {
            return i;      // 찾은 위치 반환
        }
    }
    return -1;             // 없으면 -1
}
② 이진 탐색 (Binary Search)

정렬된 배열에서 중간값과 비교하며 절반씩 탐색 범위를 줄입니다. O(log n) — 반드시 정렬 필요!

💡 핵심 아이디어

배열이 정렬되어 있으면 중간값과 비교만으로 절반을 제거할 수 있습니다. n=1,000,000 이어도 약 20번이면 충분!

이진 탐색C
#include <stdio.h>

int binary_search(int arr[], int n, int target) {
    int left = 0;
    int right = n - 1;

    while (left <= right) {
        int mid = (left + right) / 2;

        if (arr[mid] == target) return mid;
        else if (arr[mid] < target) left = mid + 1;
        else right = mid - 1;
    }
    return -1;
}
③ BFS (너비 우선 탐색)

그래프/트리에서 가까운 노드부터 탐색합니다. O(V+E) — 큐(Queue) 사용

🌊 BFS 활용
  • 최단 경로 찾기 (가중치 없는 그래프)
  • 레벨별 탐색 필요할 때
  • 연결 요소 개수 세기
BFS — 너비 우선 탐색C
#include <stdio.h>

#define MAX 100

int graph[MAX][MAX];
int visited[MAX];
int queue[MAX];
int front = 0, rear = 0;

void bfs(int start, int n) {
    visited[start] = 1;
    queue[rear++] = start;

    while (front < rear) {
        int cur = queue[front++];
        printf("%d ", cur);

        for (int next = 0; next < n; next++) {
            if (graph[cur][next] && !visited[next]) {
                visited[next] = 1;
                queue[rear++] = next;
            }
        }
    }
}
④ DFS (깊이 우선 탐색)

한 방향으로 최대한 깊게 들어간 후 돌아오는 탐색. O(V+E) — 스택/재귀 사용

DFS — 깊이 우선 탐색C
#include <stdio.h>

#define MAX 100

int graph[MAX][MAX];
int visited[MAX];

void dfs(int cur, int n) {
    visited[cur] = 1;
    printf("%d ", cur);

    for (int next = 0; next < n; next++) {
        if (graph[cur][next] && !visited[next]) {
            dfs(next, n);
        }
    }
}
🔑 BFS vs DFS 선택 기준
  • BFS: 최단 경로, 가까운 노드 먼저 탐색
  • DFS: 경로 존재 여부, 사이클 탐지, 위상 정렬

① 스택 (Stack)

LIFO(Last In First Out) — 나중에 들어온 것이 먼저 나갑니다.

📚 활용 사례
  • 괄호 검사
  • DFS 구현
  • 재귀 호출 스택
  • 수식 계산 (후위 표기)
C
#include <stdio.h>

#define MAX 100

int stack[MAX];
int top = -1;

void push(int x) {
    stack[++top] = x;
}

int pop(void) {
    return stack[top--];
}

int peek(void) {
    return stack[top];
}

int main(void) {
    push(1);
    push(2);
    push(3);

    printf("pop: %d\n", pop());    // 3
    printf("peek: %d\n", peek());  // 2
    return 0;
}
② 큐 (Queue)

FIFO(First In First Out) — 먼저 들어온 것이 먼저 나갑니다.

🎡 활용 사례
  • BFS 구현
  • 프린터 대기열
  • 프로세스 스케줄링
  • 레벨 순서 트리 탐색
C
#include <stdio.h>

#define MAX 100

int queue[MAX];
int front = 0;
int rear = 0;

void enqueue(int x) {
    queue[rear++] = x;
}

int dequeue(void) {
    return queue[front++];
}

int main(void) {
    enqueue(1);
    enqueue(2);

    printf("dequeue: %d\n", dequeue());  // 1
    return 0;
}
③ 힙 (Heap) — 우선순위 큐

완전 이진 트리 구조로 최솟값(또는 최댓값)을 O(log n)에 삽입/삭제합니다.

힙 — 우선순위 큐C
#include <stdio.h>

#define MAX 100

int heap[MAX];
int size = 0;

void push(int x) {
    int i = ++size;
    while (i > 1 && heap[i / 2] > x) {
        heap[i] = heap[i / 2];
        i /= 2;
    }
    heap[i] = x;
}

int pop(void) {
    int ret = heap[1];
    int last = heap[size--];
    int parent = 1, child = 2;

    while (child <= size) {
        if (child < size && heap[child] > heap[child + 1]) child++;
        if (last <= heap[child]) break;
        heap[parent] = heap[child];
        parent = child;
        child *= 2;
    }
    heap[parent] = last;
    return ret;
}
④ 유니온-파인드 (Union-Find)

집합의 합병과 원소의 소속 집합 확인을 효율적으로 처리합니다. 크루스칼 알고리즘의 핵심!

유니온-파인드C
#include <stdio.h>

#define MAX 100

int parent[MAX];

int find(int x) {
    if (parent[x] == x) return x;
    return parent[x] = find(parent[x]);  // 경로 압축
}

void unite(int a, int b) {
    a = find(a);
    b = find(b);

    if (a != b) parent[b] = a;
}

int main(void) {
    int n = 5;
    for (int i = 0; i < n; i++) parent[i] = i;

    unite(0, 1);
    unite(1, 2);

    printf("%s\n", find(0) == find(2) ? "YES" : "NO");
    return 0;
}

① 동전 교환 문제

거스름돈을 최소 개수의 동전으로 만들기 — 큰 단위부터 사용합니다.

동전 교환 (그리디)C
#include <stdio.h>

int main(void) {
    int coins[] = {500, 100, 50, 10};
    int n = 4;
    int money = 1260;
    int count = 0;

    for (int i = 0; i < n; i++) {
        count += money / coins[i];
        money %= coins[i];
    }

    printf("필요한 동전 개수: %d\n", count);
    return 0;
}
② 회의실 배정 문제

하나의 회의실에서 가능한 많은 회의를 배정하기 — 끝나는 시간 기준으로 정렬!

회의실 배정C
#include <stdio.h>

struct Meeting {
    int start;
    int end;
};

void sort_by_end(struct Meeting a[], int n) {
    for (int i = 0; i < n - 1; i++) {
        for (int j = i + 1; j < n; j++) {
            if (a[i].end > a[j].end ||
               (a[i].end == a[j].end && a[i].start > a[j].start)) {
                struct Meeting temp = a[i];
                a[i] = a[j];
                a[j] = temp;
            }
        }
    }
}

int main(void) {
    struct Meeting a[] = {{1,4}, {3,5}, {0,6}, {5,7}, {8,9}};
    int n = 5;
    int count = 0;
    int last_end = 0;

    sort_by_end(a, n);

    for (int i = 0; i < n; i++) {
        if (a[i].start >= last_end) {
            count++;
            last_end = a[i].end;
        }
    }

    printf("선택 가능한 회의 수: %d\n", count);
    return 0;
}

DP의 핵심 개념
💡 메모이제이션 (Memoization)

한 번 계산한 결과를 테이블에 저장해두고 다시 필요할 때 재사용합니다. "계산 결과를 기억하는 재귀"

① 피보나치 수열 — 재귀 vs DP 비교
피보나치 — 세 가지 방법C
#include <stdio.h>

// 방법 1: 순수 재귀 — O(2^n), 매우 느림
int fib_recursive(int n) {
    if (n <= 1) return n;
    return fib_recursive(n - 1) + fib_recursive(n - 2);
}

// 방법 2: 메모이제이션 — O(n)
int memo[100];
int fib_memo(int n) {
    if (n <= 1) return n;
    if (memo[n] != -1) return memo[n];
    return memo[n] = fib_memo(n - 1) + fib_memo(n - 2);
}

// 방법 3: 반복문 DP — O(n)
int fib_dp(int n) {
    int dp[100];
    dp[0] = 0;
    dp[1] = 1;

    for (int i = 2; i <= n; i++) {
        dp[i] = dp[i - 1] + dp[i - 2];
    }
    return dp[n];
}
② 0/1 배낭 문제 (Knapsack)

무게 제한 W의 배낭에 가치 합이 최대가 되도록 물건을 선택합니다. O(nW)

0/1 배낭 문제 (DP)C
#include <stdio.h>

int max(int a, int b) {
    return a > b ? a : b;
}

int knapsack(int weight[], int value[], int n, int capacity) {
    int dp[101][101] = {0};

    for (int i = 1; i <= n; i++) {
        for (int w = 0; w <= capacity; w++) {
            dp[i][w] = dp[i - 1][w];

            if (w >= weight[i - 1]) {
                dp[i][w] = max(dp[i][w],
                               dp[i - 1][w - weight[i - 1]] + value[i - 1]);
            }
        }
    }
    return dp[n][capacity];
}
③ 최장 공통 부분수열 (LCS)

두 문자열의 공통된 가장 긴 부분수열을 찾습니다. O(mn)

LCS — 최장 공통 부분수열C
#include <stdio.h>
#include <string.h>

int max(int a, int b) {
    return a > b ? a : b;
}

int lcs(char s1[], char s2[]) {
    int n = strlen(s1);
    int m = strlen(s2);
    int dp[101][101] = {0};

    for (int i = 1; i <= n; i++) {
        for (int j = 1; j <= m; j++) {
            if (s1[i - 1] == s2[j - 1]) {
                dp[i][j] = dp[i - 1][j - 1] + 1;
            } else {
                dp[i][j] = max(dp[i - 1][j], dp[i][j - 1]);
            }
        }
    }
    return dp[n][m];
}

① 다익스트라 (Dijkstra) — 최단경로

음의 가중치가 없는 그래프에서 출발점에서 모든 정점까지의 최단거리를 구합니다. O((V+E) log V)

🗺 핵심 아이디어

현재 가장 거리가 짧은 미방문 정점을 선택 → 인접 정점의 거리 갱신 → 반복. 우선순위 큐(힙)로 구현.

다익스트라C
#include <stdio.h>

#define INF 1000000000
#define MAX 100

int graph[MAX][MAX];
int dist[MAX];
int visited[MAX];

void dijkstra(int start, int n) {
    for (int i = 0; i < n; i++) dist[i] = INF;
    dist[start] = 0;

    for (int step = 0; step < n; step++) {
        int cur = -1;
        int best = INF;

        for (int i = 0; i < n; i++) {
            if (!visited[i] && dist[i] < best) {
                best = dist[i];
                cur = i;
            }
        }

        if (cur == -1) break;
        visited[cur] = 1;

        for (int next = 0; next < n; next++) {
            if (graph[cur][next] > 0 && dist[next] > dist[cur] + graph[cur][next]) {
                dist[next] = dist[cur] + graph[cur][next];
            }
        }
    }
}
② 크루스칼 (Kruskal) — 최소신장트리

간선을 가중치 기준 오름차순 정렬 후 사이클이 생기지 않는 간선을 추가합니다. O(E log E)

크루스칼 MSTC
#include <stdio.h>

#define MAX 100

struct Edge {
    int u, v, w;
};

int parent[MAX];

int find(int x) {
    if (parent[x] == x) return x;
    return parent[x] = find(parent[x]);
}

void unite(int a, int b) {
    a = find(a);
    b = find(b);
    if (a != b) parent[b] = a;
}

void sort_edges(struct Edge e[], int m) {
    for (int i = 0; i < m - 1; i++) {
        for (int j = i + 1; j < m; j++) {
            if (e[i].w > e[j].w) {
                struct Edge temp = e[i];
                e[i] = e[j];
                e[j] = temp;
            }
        }
    }
}

int kruskal(struct Edge e[], int n, int m) {
    int cost = 0;
    for (int i = 0; i < n; i++) parent[i] = i;
    sort_edges(e, m);

    for (int i = 0; i < m; i++) {
        if (find(e[i].u) != find(e[i].v)) {
            unite(e[i].u, e[i].v);
            cost += e[i].w;
        }
    }
    return cost;
}

    while (front < rear) {

        printf("%d ", cur);

투 포인터 (Two Pointers)

배열의 양 끝 또는 같은 방향에 두 포인터를 두고 조건에 따라 이동시키며 O(n²)을 O(n)으로 줄입니다.

① 두 수의 합 — 정렬된 배열
💡 아이디어

왼쪽(left=0)과 오른쪽(right=n-1) 포인터. 합이 크면 right↓, 작으면 left↑

두 수의 합 (투 포인터)C
#include <stdio.h>

int two_sum_sorted(int arr[], int n, int target) {
    int left = 0;
    int right = n - 1;

    while (left < right) {
        int sum = arr[left] + arr[right];

        if (sum == target) return 1;
        else if (sum < target) left++;
        else right--;
    }
    return 0;
}
② 연속 부분배열의 합
합이 S인 연속 부분배열C
#include <stdio.h>

int subarray_sum(int arr[], int n, int target) {
    int left = 0;
    int sum = 0;
    int count = 0;

    for (int right = 0; right < n; right++) {
        sum += arr[right];

        while (sum > target && left <= right) {
            sum -= arr[left++];
        }

        if (sum == target) count++;
    }
    return count;
}

#include <stdio.h>

int max_subarray_k(int arr[], int n, int k) {
    int sum = 0;
    for (int i = 0; i < k; i++) sum += arr[i];

    int answer = sum;
    for (int i = k; i < n; i++) {
        sum += arr[i];
        sum -= arr[i - k];
        if (sum > answer) answer = sum;
    }
    return answer;
}
최장 증가 부분수열 (LIS)

수열에서 오름차순으로 증가하는 가장 긴 부분수열의 길이를 구합니다.

🔑 점화식

dp[i] = max(dp[j]+1) for all j < i where arr[j] < arr[i]

단순 DP: O(n²) / 이진탐색 최적화: O(n log n)

LIS — 최장 증가 부분수열C
#include <stdio.h>

int lis_dp(int arr[], int n) {
    int dp[100];
    int answer = 1;

    for (int i = 0; i < n; i++) dp[i] = 1;

    for (int i = 1; i < n; i++) {
        for (int j = 0; j < i; j++) {
            if (arr[j] < arr[i] && dp[i] < dp[j] + 1) {
                dp[i] = dp[j] + 1;
            }
        }
        if (answer < dp[i]) answer = dp[i];
    }
    return answer;
}

플로이드-워샬 (Floyd-Warshall)

모든 정점 쌍 사이의 최단거리를 한 번에 구합니다. O(V³) — V가 작을 때 사용

💡 핵심 점화식

dist[i][j] = min(dist[i][j], dist[i][k] + dist[k][j])

중간 정점 k를 모든 정점에 대해 순서대로 고려합니다.

플로이드-워샬C
#include <stdio.h>

#define INF 1000000000
#define MAX 100

int dist[MAX][MAX];

void floyd_warshall(int n) {
    for (int k = 0; k < n; k++) {
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < n; j++) {
                if (dist[i][k] != INF && dist[k][j] != INF &&
                    dist[i][j] > dist[i][k] + dist[k][j]) {
                    dist[i][j] = dist[i][k] + dist[k][j];
                }
            }
        }
    }
}
벨만-포드 (Bellman-Ford)

음의 가중치가 있는 그래프의 최단경로 탐색. 음의 사이클 검출 가능. O(VE)

⚠ 다익스트라 vs 벨만-포드
  • 다익스트라: 음의 가중치 ❌, 빠름 O((V+E)logV)
  • 벨만-포드: 음의 가중치 ✅, 느림 O(VE), 음수 사이클 검출
벨만-포드C
#include <stdio.h>

#define INF 1000000000
#define MAXE 1000

struct Edge {
    int from, to, cost;
};

int dist[101];

int bellman_ford(struct Edge edges[], int n, int m, int start) {
    for (int i = 1; i <= n; i++) dist[i] = INF;
    dist[start] = 0;

    for (int i = 1; i <= n - 1; i++) {
        for (int j = 0; j < m; j++) {
            int u = edges[j].from;
            int v = edges[j].to;
            int w = edges[j].cost;

            if (dist[u] != INF && dist[v] > dist[u] + w) {
                dist[v] = dist[u] + w;
            }
        }
    }

    for (int j = 0; j < m; j++) {
        int u = edges[j].from;
        int v = edges[j].to;
        int w = edges[j].cost;
        if (dist[u] != INF && dist[v] > dist[u] + w) return 0;
    }
    return 1;
}

💡 핵심 아이디어

실패 함수(failure function)를 미리 계산하여, 불일치 발생 시 처음부터 다시 비교하지 않고 적절한 위치로 점프합니다.

KMP 알고리즘C
#include <stdio.h>
#include <string.h>

void make_pi(char pattern[], int pi[]) {
    int m = strlen(pattern);
    int j = 0;

    for (int i = 1; i < m; i++) {
        while (j > 0 && pattern[i] != pattern[j]) {
            j = pi[j - 1];
        }
        if (pattern[i] == pattern[j]) {
            pi[i] = ++j;
        }
    }
}

void kmp(char text[], char pattern[]) {
    int n = strlen(text);
    int m = strlen(pattern);
    int pi[100] = {0};
    int j = 0;

    make_pi(pattern, pi);

    for (int i = 0; i < n; i++) {
        while (j > 0 && text[i] != pattern[j]) {
            j = pi[j - 1];
        }
        if (text[i] == pattern[j]) {
            if (j == m - 1) {
                printf("패턴 발견 위치: %d\n", i - m + 1);
                j = pi[j];
            } else {
                j++;
            }
        }
    }
}
